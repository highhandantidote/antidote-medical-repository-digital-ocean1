        thread.title = request.form['title']
        thread.category = request.form['category']
        db.session.commit()
        return redirect(url_for('web.community_dashboard'))

document.addEventListener('DOMContentLoaded', () => {
    console.log('DOM loaded');
    const createButton = document.querySelector('.create-thread');
    if (createButton) {
        createButton.addEventListener('click', () => {
            console.log('Create Thread clicked');
            window.location.href = '/dashboard/community/create';
        });
    } else {
        console.log('Create Thread button not found');
    }

    document.querySelectorAll('.actions button').forEach(button => {
        button.addEventListener('click', (e) => {
            console.log('Action clicked:', e.target.textContent, 'Thread ID:', button.closest('tr')?.dataset.threadId);
            const action = e.target.textContent.toLowerCase();
            const threadId = button.closest('tr')?.dataset.threadId;
            if (threadId) {
                if (action === 'view') window.location.href = `/community/thread/${threadId}`;
                else if (action === 'edit') window.location.href = `/dashboard/community/edit/${threadId}`;
                else if (action === 'delete') {
                    if (confirm('Are you sure?')) {
                        fetch(`/api/community/${threadId}`, { method: 'DELETE' })
                            .then(response => response.json())
                            .then(data => location.reload());
                    }
                }
            } else {
                console.log('Thread ID not found for action:', action);
            }
        });
    });
});
